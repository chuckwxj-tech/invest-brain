// 優先度天井（PCP）準拠 非フェルミ適応型MBL保護ハンドラ（最速割込対応）
static irqreturn_t kut_hal_nfl_emergency_isr(int irq, void *dev_id)
{

    u32 temp_raw;
    u32 shaped_mbl_voltage;
    u32 counter_holonomy_val;

#elif defined(__aarch64__)
